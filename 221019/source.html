using System;
using System.Collections;
using System.Collections.Generic;
using UnityEngine;
 
public class GameManager : Singletone<GameManager>
{
    private void Awake()
    {
        Application.targetFrameRate = 60;
        Data.GetMyStat(); // 유저 스탯정보 가져오기
    }
 
    void Start()
    {
        StartCoroutine(AutoSave());  // 자동저장 실행
        TutorialManager.inst.Setup(); // 튜토리얼정보 셋업
        StartCoroutine(TutorialManager.inst.TutoBeforeAction(StartNext)); // 튜토리얼 진행전에 실행할 액션
    }
 
    private void Update()
    {
        if (Input.GetKeyDown(KeyCode.Escape))
        {
            // backKeyStack
            BackKeyAction();
        }
    }
 
 
 
    [Header("뒤로가기키")]
    public Stack<Func<bool>> backKeyStack = new Stack<Func<bool>>();
    private void BackKeyAction()
    {
        if (backKeyStack.Count > 0)
        {
            if (backKeyStack.Pop()() == false)
            {
                BackKeyAction();
            }
        }
        else
        {
            PopupRef.inst.dialogpop
            .SetTitle("게임종료")
            .SetContext("게임을 종료 하시겠습니까?")
            .SetBtnCol(DialogBtnCol.Green, DialogBtnCol.Gray)
            .SetAction(()=>{
                Data.LocalSave();
                Application.Quit();
            },()=>{})
            .Show();
        }
    }
    public void AddBackKeyAction(Func<bool> action)
    {
        backKeyStack.Push(action);
    }
 
 
 
    void StartNext()
    {
        // 튜토리얼 선행 액션이 실행된후 진행될 일들
        TutoCanvas.inst.SetBig(); // 처음엔 튜토리얼 화면크게
        Spawner.inst.Add(SpawnType.Normal); // 일반몬스터 소환되게 설정함
        myPlayer.state = Entity.State.Attack; // 플레이어가 공격가능하게
    }
 
    // 클릭시 서버에 저장하는 함수
    public void _OnClickServerSave()
    {
        Data.ServerSave();
    }
 
 
    // 게임매니저 단에서 코루틴 돌리는 함수
    public void NewCoroutine(IEnumerator coroutine)
    {
        StartCoroutine(coroutine);
    }
 
 
    // 일정 시간마다 게임 로컬저장 자동으로 하는 함수
    private IEnumerator AutoSave()
    {
        Data.autosavetime = Data.time.AddSeconds(60);
        while (true)
        {
            yield return new WaitUntil(() => Data.time >= Data.autosavetime);
            Data.LocalSave();
        }
    }
 
    public void _OnClickLocalSave()
    {
        Data.LocalSave();
    }
  
 
 
    [Header("플레이어캐릭터")]
    public PlayerEntity myPlayer;
 
    [Header("색상참조")]
    public Colors colors = new Colors();
    
 
}

    public List<Color> gradeColors = new List<Color>();
